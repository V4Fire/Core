Optional concatArray

concatArray: undefined | false | true
  • Defined in index.d.ts:237

If true, then for merging two arrays will be used a concatenation strategy (works only with the "deep" mode)

default

false

example
// {a: [2]}
Object.mixin({deep: true, concatArray: false}, {a: [1]}, {a: [2]});

// {a: [1, 2]}
Object.mixin({deep: true, concatArray: true}, {a: [1]}, {a: [2]});

Optional deep

deep: undefined | false | true
  • Defined in index.d.ts:123

If true, then object properties are copied recursively

default

false

example
// {a: {c: 2}}
Object.mixin({deep: false}, {a: {b: 1}}, {a: {c: 2}});

// {a: {b: 1, c: 2}}
Object.mixin({deep: true}, {a: {b: 1}}, {a: {c: 2}});

Optional withProto

withProto: undefined | false | true
  • Defined in index.d.ts:221

If true, then object properties is copied with their prototypes (works only with the "deep" mode)

default

false

example
const proto = {
  a: {
    b: 2
  },

  c: 3
};

const obj = Object.create(proto);
Object.mixin({deep: true, withProto: false}, obj, {c: 2, a: {d: 4}});

// 2
// 4
// 2
// true
// true
console.log(
  obj.c,
  obj.a.d,
  obj.a.b,
  obj.a.hasOwnProperty('d'),
  obj.a.hasOwnProperty('b')
);

const obj2 = Object.create(proto);
Object.mixin({deep: true, withProto: true}, obj2, {c: 2, a: {d: 4}});

// 2
// 4
// 2
// true
// false
console.log(
  obj2.c,
  obj2.a.d,
  obj2.a.b,
  obj2.a.hasOwnProperty('d'),
  obj2.a.hasOwnProperty('b')
);

Optional withUndef

withUndef: undefined | false | true
  • Defined in index.d.ts:159

If true, then the original value of an object property can be rewritten from another object with undefined value

default

false

example
// {a: 1}
Object.mixin({withUndef: false}, {a: 1}, {a: undefined});

// {a: undefined}
Object.mixin({withUndef: true}, {a: 1}, {a: undefined});
