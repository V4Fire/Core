Module src/core/queue

core/queue

This module provides an abstract class for the Queue data structure. The submodules contain different classes and interfaces that extends or implements that class.

The main module re-exports these implementations:

  • AbstractQueue — an alias for core/queue/interface/Queue;
  • AbstractWorkerQueue — an alias for core/queue/worker/interface/WorkerQueue;
  • Queue — an alias for core/queue/simple;
  • OrderedQueue — an alias for core/queue/order;
  • MergeQueue — an alias for core/queue/merge.
  • AbstractWorkerQueue — an alias for core/queue/worker.
  • WorkerQueue — an alias for core/queue/worker/simple.
  • MergeWorkerQueue — an alias for core/queue/worker/merge.
import { AbstractQueue } from 'core/queue';

export default class Queue extends AbstractQueue {
  internalQueue = [];

  get head() {
    return this.internalQueue[0];
  }

  get length() {
    return this.internalQueue.length;
  }

  push(el) {
    return this.internalQueue.push(el);
  }

  pop() {
    return this.internalQueue.shift();
  }

  clear() {
    this.internalQueue = [];
  }
}
