Module src/core/status-codes

core/status-codes

This module provides a map of HTTP status codes.

import StatusCodes from 'core/status-codes';

// 301
console.log(StatusCodes.MOVED_PERMANENTLY);

// 404
console.log(StatusCodes.NOT_FOUND);
