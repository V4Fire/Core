# lang

This module provides a table of texts to internalize the application.

```js
export default {
  ru: {
    "my-component": {
      "time": "время",
      "{count} product": [
        "{count} продукт",
        "{count} продукта",
        "{count} продуктов",
        "{count} продуктов"
      ],
      "my name is {name}": "меня зовут {name}",
      "apple": "яблоко"
    }
  },
  en: {
    "my-component": {
      "time": "time",
      "{count} product": [
        "{count} product",
        "{count} products",
        "{count} products",
        "{count} products"
      ],
      "my name is {name}": "my name is {name}",
      "apple": "apple"
    }
  }
};
```
